#+BEGIN_SRC python
SOLR_URL = os.environ.get("SOLR_URL")
#+END_SRC

** Init the connection

#+BEGIN_SRC python
solr = pysolr.Solr(SOLR_URL, always_commit=True)
#+END_SRC

** Add documents

#+BEGIN_SRC python
solr.add([

    },
    {

        "title": "The Banana: Tasty or Dangerous?",

            { "id": "child_doc_1", "title": "peel" },
            { "id": "child_doc_2", "title": "seed" },
        ]
    },
])
#+END_SRC
